 * @version 1.0
 * @requires jQuery
 */

(function ($) {
	$.wCheckImage = function (url, success, failure) {
		var img = new Image(),
			loaded = false,
			errored = false,
			errors = {};

		img.onload = function () {
			if (loaded) { return; }
			loaded = true;

			if (success && success.call) {
				success.call(img);
			}
		};

		img.onerror = function () {
			if (errored) { return; }
			errors[url] = errored = true;

			if (failure && failure.call) {
				failure.call(img);
			}
		};

		if (errors[url]) {
			failure.call(img);
			return;
		}

		img.src = url;

		if (img.complete) {
			success.call(img);
		}
	};
}(jQuery));
